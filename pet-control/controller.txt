// Pet
// Summon pet
Hotkey ( LCtrl RCtrl E )
{
   sendto (pbae1 pbae2 pbae3)
   Slash ("/qbar 9 1")
   wait(200)
   Key ( 4 ) // Summon Pet
   wait(200)
   Slash ("/qbar 1 1")
}
